23  import java.io.ByteArrayOutputStream;
24  import java.io.File;
25  import java.io.IOException;
26  import java.io.OutputStreamWriter;
27  import java.nio.charset.Charset;
28  import java.util.List;
29  import java.util.Locale;

31  import static com.google.common.collect.Iterables.toArray;
32  import static java.lang.System.nanoTime;
33  import static java.util.Arrays.asList;
34  import static java.util.Collections.singleton;
35  import static javax.tools.Diagnostic.Kind.MANDATORY_WARNING;
36  import static javax.tools.Diagnostic.Kind.WARNING;
37  import static javax.tools.JavaCompiler.CompilationTask;
38  import static javax.tools.StandardLocation.CLASS_OUTPUT;
39  import static javax.tools.StandardLocation.SOURCE_OUTPUT;

41  /**
42   * Java compiler for testing.
43   *
44   * @author <a href="http://github.com/nicoulaj">Julien Nicoulaud</a>
45   */
46  public final class JavaCompilationTester {
47  
48      private static final JavaCompiler COMPILER = ToolProvider.getSystemJavaCompiler();
49  
50      private static final File COMPILATION_OUTPUT_BASE_DIRECTORY = new File("target", "test-compilation");
51  
52      public Report compile(File source, String... options) {
53          return compile(source.toString(), options);
54      }
55  
56      public Report compile(String source, String... options) {
57          return compile(asList(source), asList(options));
58      }
59  
60      public Report compile(Iterable<String> sources, Iterable<String> options) {
61  
62          for (String option : options)
63              if (COMPILER.isSupportedOption(option) < 0)
64                  throw new IllegalArgumentException("Unsupported option \"" + option + "\"");
65  
66          final DiagnosticCollector<JavaFileObject> collector = new DiagnosticCollector<JavaFileObject>();
67          final StandardJavaFileManager fileManager = COMPILER.getStandardFileManager(collector, Locale.getDefault(), Charset.defaultCharset());
68  
69          File outputClasses;
70          File outputSources;
71          try {
72              final File output = new File(COMPILATION_OUTPUT_BASE_DIRECTORY, String.valueOf(nanoTime()));
73              if (!output.mkdirs()) throw new RuntimeException("Failed creating compilation output directory");
74  
75              outputClasses = new File(output, "classes");
76              if (!outputClasses.mkdir()) throw new RuntimeException("Failed creating compilation output directory");
77              fileManager.setLocation(CLASS_OUTPUT, singleton(outputClasses));
78  
79              outputSources = new File(output, "sources");
80              if (!outputSources.mkdir()) throw new RuntimeException("Failed creating compilation output directory");
81              fileManager.setLocation(SOURCE_OUTPUT, singleton(outputClasses));
82  
83          } catch (IOException e) {
84              throw new RuntimeException("Failed setting up compilation output directories", e);
85          }
86  
87          ByteArrayOutputStream stream = new ByteArrayOutputStream();
88          OutputStreamWriter out = new OutputStreamWriter(stream);
89          try {
90  
91              final CompilationTask task = COMPILER.getTask(out,
92                                                            fileManager,
93                                                            collector,
94                                                            options,
95                                                            null,
96                                                            fileManager.getJavaFileObjects(toArray(sources, String.class)));
97  
98              final Boolean successful = task.call();
99  
100             final String stdout = new String(stream.toByteArray());
101 
102             return new Report(outputClasses, outputSources, successful, collector.getDiagnostics(), stdout);
103         } finally {
104             try {
105                 stream.close();
106                 out.close();
107             } catch (IOException e) {
108                 throw new RuntimeException(e);
109             }
110         }
111     }
112 
113     public static final class Report {
114 
115         private final File classesDirectory;
116 
117         private final File sourcesDirectory;
118 
119         private final boolean successful;
120 
121         private final List<Diagnostic<? extends JavaFileObject>> diagnostics;
122 
123         private final String stdout;
124 
125         Report(final File classesDirectory,
126                final File sourcesDirectory,
127                final boolean successful,
128                final List<Diagnostic<? extends JavaFileObject>> diagnostics,
129                final String stdout) {
130             this.classesDirectory = classesDirectory;
131             this.sourcesDirectory = sourcesDirectory;
132             this.successful = successful;
133             this.diagnostics = diagnostics;
134             this.stdout = stdout;
135         }
136 
137         public File getClassesDirectory() {
138             return classesDirectory;
139         }
140 
141         public File getSourcesDirectory() {
142             return sourcesDirectory;
143         }
144 
145         public boolean isSuccessful() {
146             return successful;
147         }
148 
149         public boolean isFailed() {
150             return !successful;
151         }
152 
153         public boolean hasErrors() {
154             return !successful;
155         }
156 
157         public boolean hasWarnings() {
158             for (Diagnostic<? extends JavaFileObject> diagnostic : diagnostics)
159                 if (diagnostic.getKind() == WARNING || diagnostic.getKind() == MANDATORY_WARNING)
160                     return true;
161             return false;
162         }
163 
164         public List<Diagnostic<? extends JavaFileObject>> getDiagnostics() {
165             return diagnostics;
166         }
167 
168         public String getStdout() {
169             return stdout;
170         }
171     }
